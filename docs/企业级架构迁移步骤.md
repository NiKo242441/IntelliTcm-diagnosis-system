## 当前项目结构分析

```
当前项目结构:
├── TcmAiDiagnosis.Entities/      # 实体层（30+实体文件）
├── TcmAiDiagnosis.EFContext/     # 数据访问层
```

#### 步骤1.3：创建仓储实现

**示例：创建 PatientRepository**

```csharp
// TcmAiDiagnosis.Infrastructure/Persistence/Repositories/PatientRepository.cs
using Microsoft.EntityFrameworkCore;
using TcmAiDiagnosis.Domain.Entities;
using TcmAiDiagnosis.Domain.Interfaces;
using TcmAiDiagnosis.EFContext;

namespace TcmAiDiagnosis.Infrastructure.Persistence.Repositories
{
    /// <summary>
    /// 患者仓储实现
    /// </summary>
    public class PatientRepository : IPatientRepository
    {
        private readonly TcmAiDiagnosisContext _context;

        public PatientRepository(TcmAiDiagnosisContext context)
        {
            _context = context;
        }

        public async Task<Patient> GetByIdAsync(int id)
        {
            return await _context.Patients
                .Include(p => p.User)
                .FirstOrDefaultAsync(p => p.Id == id);
        }

        public async Task<List<Patient>> GetAllAsync(int pageIndex, int pageSize)
        {
            return await _context.Patients
                .Include(p => p.User)
                .Skip((pageIndex - 1) * pageSize)
                .Take(pageSize)
                .ToListAsync();
        }

        public async Task<int> GetCountAsync()
        {
            return await _context.Patients.CountAsync();
        }

        public async Task AddAsync(Patient patient)
        {
            await _context.Patients.AddAsync(patient);
            await _context.SaveChangesAsync();
        }

        public async Task UpdateAsync(Patient patient)
        {
            _context.Patients.Update(patient);
            await _context.SaveChangesAsync();
        }

        public async Task DeleteAsync(int id)
        {
            var patient = await GetByIdAsync(id);
            if (patient != null)
            {
                _context.Patients.Remove(patient);
                await _context.SaveChangesAsync();
            }
        }
    }
}
```

#### 步骤2.3：创建应用服务示例

```csharp
// TcmAiDiagnosis.Application/Services/PatientAppService.cs
using TcmAiDiagnosis.Application.DTOs.Patients;
using TcmAiDiagnosis.Application.Interfaces;
using TcmAiDiagnosis.Domain.Entities;
using TcmAiDiagnosis.Domain.Interfaces;

namespace TcmAiDiagnosis.Application.Services
{
    /// <summary>
    /// 患者应用服务
    /// </summary>
    public class PatientAppService : IPatientAppService
    {
        private readonly IPatientRepository _patientRepository;
        private readonly IMapper _mapper;

        public PatientAppService(
            IPatientRepository patientRepository,
            IMapper mapper)
        {
            _patientRepository = patientRepository;
            _mapper = mapper;
        }

        /// <summary>
        /// 获取患者列表
        /// </summary>
        public async Task<PatientListDto> GetPatientListAsync(int pageIndex, int pageSize)
        {
            var patients = await _patientRepository.GetAllAsync(pageIndex, pageSize);
            var totalCount = await _patientRepository.GetCountAsync();

            return new PatientListDto
            {
                Patients = _mapper.Map<List<PatientDto>>(patients),
                TotalCount = totalCount,
                PageIndex = pageIndex,
                PageSize = pageSize
            };
        }

        /// <summary>
        /// 获取患者详情
        /// </summary>
        public async Task<PatientDto> GetPatientByIdAsync(int id)
        {
            var patient = await _patientRepository.GetByIdAsync(id);
            if (patient == null)
                throw new NotFoundException($"患者 {id} 不存在");

            return _mapper.Map<PatientDto>(patient);
        }

        /// <summary>
        /// 创建患者
        /// </summary>
        public async Task<PatientDto> CreatePatientAsync(CreatePatientInput input)
        {
            var patient = _mapper.Map<Patient>(input);
            patient.CreatedAt = DateTime.Now;
            patient.UpdatedAt = DateTime.Now;

            await _patientRepository.AddAsync(patient);

            return _mapper.Map<PatientDto>(patient);
        }

        /// <summary>
        /// 更新患者
        /// </summary>
        public async Task<PatientDto> UpdatePatientAsync(int id, UpdatePatientInput input)
        {
            var patient = await _patientRepository.GetByIdAsync(id);
            if (patient == null)
                throw new NotFoundException($"患者 {id} 不存在");

            _mapper.Map(input, patient);
            patient.UpdatedAt = DateTime.Now;

            await _patientRepository.UpdateAsync(patient);

            return _mapper.Map<PatientDto>(patient);
        }

        /// <summary>
        /// 删除患者
        /// </summary>
        public async Task DeletePatientAsync(int id)
        {
            await _patientRepository.DeleteAsync(id);
        }
    }
}
```

```csharp
// TcmAiDiagnosis.Web/Middleware/GlobalExceptionMiddleware.cs
using TcmAiDiagnosis.Domain.Exceptions;

namespace TcmAiDiagnosis.Web.Middleware
{
    /// <summary>
    /// 全局异常处理中间件
    /// </summary>
    public class GlobalExceptionMiddleware
    {
        private readonly RequestDelegate _next;
        private readonly ILogger<GlobalExceptionMiddleware> _logger;

        public GlobalExceptionMiddleware(RequestDelegate next, ILogger<GlobalExceptionMiddleware> logger)
        {
            _next = next;
            _logger = logger;
        }

        public async Task InvokeAsync(HttpContext context)
        {
            try
            {
                await _next(context);
            }
            catch (NotFoundException ex)
            {
                _logger.LogWarning(ex, "资源未找到");
                context.Response.StatusCode = 404;
                await context.Response.WriteAsJsonAsync(new { error = ex.Message });
            }
            catch (BusinessRuleException ex)
            {
                _logger.LogWarning(ex, "业务规则违反");
                context.Response.StatusCode = 400;
                await context.Response.WriteAsJsonAsync(new { error = ex.Message });
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "系统异常");
                context.Response.StatusCode = 500;
                await context.Response.WriteAsJsonAsync(new { error = "系统错误，请稍后重试" });
            }
        }
    }
}

// 在Program.cs中注册
app.UseMiddleware<GlobalExceptionMiddleware>();
```

## 注意事项

### 1. 渐进式迁移
- 不要一次性重构所有代码
- 优先迁移核心业务模块（患者、就诊、治疗）
- 保持旧代码可运行，逐步替换

### 2. 测试策略
- 每完成一个模块都要进行单元测试
- 保持API兼容性（如果有外部调用）
- 使用Postman测试API接口

### 3. 数据库迁移
- EF Core迁移保持不变
- 仓储层只负责CRUD操作
- 业务逻辑放在应用层或领域层

### 4. 配置管理
- 保持`appsettings.json`结构
- 敏感信息使用环境变量
- 不同环境使用不同配置文件

## 常见问题

### Q1: 现有代码如何迁移？
**A**: 采用渐进式迁移，先创建新项目，然后逐步将现有代码移动到新结构中。

### Q2: 需要重写所有代码吗？
**A**: 不需要。领域实体和数据库访问代码可以复用，主要是重新组织结构。

### Q3: 迁移期间如何保证系统正常运行？
**A**: 保持旧项目不变，新功能在新结构中开发，逐步迁移旧功能。

### Q4: 自动化迁移工具推荐？
**A**: 可以使用 ReSharper 或 Rider 的 Refactoring 功能来辅助迁移。
